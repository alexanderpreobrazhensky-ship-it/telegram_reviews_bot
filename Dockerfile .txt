FROM python:3.11-slim

WORKDIR /app

# 1. Устанавливаем ТОЛЬКО torch (самый простой способ)
RUN pip install torch==2.2.0 --index-url https://download.pytorch.org/whl/cpu

# 2. Копируем requirements.txt и устанавливаем остальное
COPY requirements.txt .
RUN pip install -r requirements.txt

# 3. Копируем код
COPY . .

# 4. Запускаем
CMD ["python", "main.py"]
